```
; multi-segment executable file template.
data segment
    ; add your data here!
    pkey db "press any key...$"
  ; 矩阵的尺寸
  RECT_WIDTH = 10
  RECT_HEIGHT = 20
  ; 矩阵初始位置
  RECT_X_INIT = - RECT_WIDTH    ;  在最左边，不显示
  RECT_Y_INIT = 10 
  ; 矩阵现在的位置
  RECT_X DW RECT_X_INIT    
  RECT_Y DW RECT_Y_INIT
  ; 颜色
  RECT_COLOR DB 100  ; 矩阵的颜色
  BACK_COLOR DB 0    ; 背景的颜色
  ; 速度的
  MOV_STEP = 5       ; 每次移动5个像素，
  ; 分辨率
  SCREEN_WIDTH = 640 ; 	
  SCREEN_HEIGHT = 480; 
  ; 休眠时间，cx是高位,dx是低位，单位是微秒
  SLEEP_CX = 1   ; 
  SLEEP_DX = 0   ;

data ends

stack segment stack
    dw   128  dup(0)
stack ends

code segment
  ASSUME CS:code
start:
; set segment registers:
    mov ax, data
    mov ds, ax
    mov es, ax

    ; add your code here
  ; 思路分析，首先设置显示模式是图形模式
  ; 然后是调用竖线显示，
  ; 然后是计算下一步的坐标。
  ; 跳转到竖线显示
  ; 这里是设置显示模式
  MOV AH , 00h
  MOV AL , 13h          ; 图形模式下，分辨率是640*480
  INT 10h
  ; 显示的。
  SHOW_MOVE:
    MOV CX , MOV_STEP    ; 循环计数
  SHOW_MOVE_LOOP:
    ; 先右边的显示
    PUSH CX
    MOV CX , RECT_X      ; 取得当前x坐标
    ADD CX , RECT_WIDTH  ; 加上宽度就是右边的x坐标
    MOV AL , RECT_COLOR
    CALL DRAW_VERTICAL 
    ; 然后左边的消除
    MOV CX , RECT_X
    MOV AL , BACK_COLOR
    CALL DRAW_VERTICAL 
    ; 下一个位置
    CALL RECT_MOVE       ; 
    ; 延时功能，
    CALL WAITP
    
    POP CX
    loop SHOW_MOVE_LOOP  ; 循环
  
  JMP SHOW_MOVE            ; 跳转

    lea dx, pkey
    mov ah, 9
    int 21h        ; output string at ds:dx
    
    ; wait for any key....    
    mov ah, 1
    int 21h
    
    mov ax, 4c00h ; exit to operating system.
    int 21h    


DRAW_VERTICAL PROC
  ; 绘制竖线的宏，参数cx是X坐标。参数AL是像素值
  PUSH AX
  PUSH BX
  PUSH CX
  PUSH DX
  MOV DX , RECT_Y              ; 行在变
  MOV BH , 0                   ; 页码
  MOV AH , 0CH                 ; 写像素功能
  ; 这里有特殊情况，超出边界了，就不用绘制
  CMP CX , 0
  JL DRAW_VERTICAL_EXIT       ; 小于0就不用绘制
  ;
  CMP CX , SCREEN_WIDTH       ; 屏幕
  JGE DRAW_VERTICAL_EXIT      ; 大于等于也不用绘制
  ;
  ADD DX , rect_height        ; 先最大
  DRAW_VERTICAL_LOOP:
    INT 10H                 ; 真正操作
    DEC DX                  ; 下一个
    CMP DX , RECT_Y         ; 判断循环条件
    JGE DRAW_VERTICAL_LOOP  ; 大于等于循环,这里是条件循环。
  DRAW_VERTICAL_EXIT:
  POP DX
  POP CX
  POP BX
  POP AX
  RET
draw_vertical ENDP


RECT_MOVE PROC
  ; 下一个位置
  PUSH AX 
  INC RECT_X               ; 下一个像素
  MOV AX , RECT_X          ; 
  CMP AX , SCREEN_WIDTH    ; 是否超出屏幕边界
  JL RECT_MOVE_EXIT        ; 小于才是在边界内
  MOV AX , RECT_X_INIT
  MOV RECT_X , RECT_X_INIT ; 重置
  RECT_MOVE_EXIT:
  POP AX
  RET
RECT_MOVE ENDP

WAITP PROC
  ; 延时功能，
  PUSH AX 
  PUSH CX
  PUSH DX
  mov ah , 86h
    mov cx , SLEEP_CX  ; 高位 ， 单位是微秒
    mov dx , SLEEP_DX  ; 低位
    int 15h
  POP DX
  POP CX
  POP AX
  RET
WAITP ENDP


code ends
end start ; set entry point and stop the assembler.

```
