```
datas segment
    ; add your data here!
    pkey db 'press any key ...$'     
    ListSize = ($ - pkey)
datas ends   ; 

stacksg segment stack
    dw   128  dup(0)
stacksg ends

codes segment     
ASSUME CS:CODES,DS:DATAS,SS:stacksg
start:
; set segment registers:    
  ; 先设置堆栈
  mov ax, stacksg
    mov ss, ax
    mov sp, 128
  ; 再设置其他寄存器。
    mov ax, datas
    mov ds, ax
    mov es, ax
  
    lea dx, pkey
    mov ah, 9
    int 21h        ; output string at ds:dx
  
  ; wait for any key....    
    mov ah, 1
    int 21h
    
    mov ax, 4c00h ; exit to operating system.
    int 21h  

COUTENTER PROC
	PUSH AX
	PUSH DX
	 ;回车换行
    MOV AH,02H
	MOV DL,0DH
	INT 21H
	MOV AH,02H
	MOV DL,0AH
	INT 21H
	POP DX
	POP AX
	RET
COUTENTER ENDP

    
codes ends

end start ; set entry point and stop the assembler.
```
