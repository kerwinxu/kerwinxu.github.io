```
print16 PROC
    ; 打印16位数字。
    ; 核心思路是不断的除以10，然后将余数放在堆栈中，最后从堆栈中弹出数字，并打印，
    ; 我这里用cx来计数
    push ax
    push bx
    push cx
    push dx

    mov cx , 0
    mov bx , 10 ; 被除数。
    print161:
        mov dx , 0 ; 
        div bx     ; 除以10，商 ax , 余数 dx
        push dx  
        inc cx 
        ; 判断是否结束，
        cmp ax , 0 
        jne print161
    ; 如下是打印
    print162:
        ; 
        pop dx ; 弹出一个
        add dl , '0' ; 数字转成ascii
        mov ah , 02H
        int 21H
        loop print162

    pop dx
    pop cx
    pop bx
    pop ax
    ret

print16 endp
```

 

```
print8 proc
    push ax 
    push bx
    push cx
    push dx
    ; 这里要打印这个数字
    ; 算法是首先从低位开始，一个一个的压入堆栈，最后从堆栈中打印就可以了（堆栈的性质，先进后出，这样就先打印数字的高位了）
    mov cx , 0 ; 这个当作计数，这个数字有多少位
    mov bl , 10 ; 被除数。
    
    print1:
        mov ah , 0  ; ax 的高位置零，原先是余数。
        ; 这里每一次都除以10,ax/bl=余数存放在AH, 商存放在AL中，然后将余数放到堆栈中
        div bl
        ; 堆栈是保存16位的，这里先将余数放在dx中，    
        mov dx , 0
        mov dl , ah 
        push dx 
        ; 数字位数计数
        inc cx 
        ; 当商为0表示结束。
        cmp al , 0 
        jne print1
    ; 这里表示取得了所有的数字了，然后就是打印了
    print2:
        ; 依旧是循环，cx的值在上一步计算出来了。
        ; 每次打印一个数字，因为堆栈的性质，这里第一个打印的反倒是数字的最高位。
        pop dx ; 弹出一个
        add dl , '0' ; 数字转成ascii
        mov ah , 02H
        int 21H
        loop print2
    pop dx
    pop cx
    pop bx
    pop ax
    ret
print8 endp
```
