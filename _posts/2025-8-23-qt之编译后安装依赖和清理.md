这个是qmake方式，也就是pro文件，我这里是win平台
```
win32{
    DEPLOY_TOOL = $${dirname(QMAKE_QMAKE)}/windeployqt
    DEPLOY_TOOL = $$replace(DEPLOY_TOOL, /,\\)
    CONFIG(debug, debug|release){
        DEPLOY_TARGET = $${OUT_PWD}/debug/$${TARGET}.exe
        TARGET_DIR = $${OUT_PWD}/debug
    }else{
        DEPLOY_TARGET = $${OUT_PWD}/release/$${TARGET}.exe
        TARGET_DIR = $${OUT_PWD}/release
    }
    DEPLOY_TARGET = $$replace(DEPLOY_TARGET, /,\\)
    TARGET_DIR = $$replace(TARGET_DIR, /,\\)
    message("$${DEPLOY_TOOL} $${DEPLOY_TARGET}")
    message("target dir: $${TARGET_DIR}")
    # &&  copy $$shell_path($$PWD/db.db) $$shell_path($$DESTDIR)
    QMAKE_POST_LINK+=$$DEPLOY_TOOL $$DEPLOY_TARGET && \
                        del /Q $$TARGET_DIR\\*.h && \
                        del /Q $$TARGET_DIR\\*.cpp && \
                        del /Q $$TARGET_DIR\\*.o && \
                        del /Q $$TARGET_DIR\\D3Dcompiler_47.dll && \
                        del /Q $$TARGET_DIR\\opengl32sw.dll
}
```
