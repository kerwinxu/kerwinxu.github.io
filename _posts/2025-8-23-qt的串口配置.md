常用的配置如下
```c
#include <QtSerialPort/QSerialPort>
#include <QtSerialPort/QSerialPortInfo>
// 如下是串口的参数
int baudrates []= {2400,4800,9600,19200,38400,57600,115500};
QString parities [] = {"No", "Even", "Odd"};
QSerialPort::Parity parities_value []={QSerialPort::NoParity, QSerialPort::EvenParity, QSerialPort::OddParity};
QString databitss [] = {"5", "6", "7", "8"};
QSerialPort::DataBits databits_value []= {QSerialPort::Data5,QSerialPort::Data6,QSerialPort::Data7, QSerialPort::Data8};
QString stopbitss [] = {"1", "1.5", "2"};
QSerialPort::StopBits stopbits_value [] = {QSerialPort::OneStop, QSerialPort::OneAndHalfStop, QSerialPort::TwoStop};
```
