```python
import torch
import torch.nn as nn  # 神经网络
import torch.optim as optim  # 优化器
from torch.utils.data import Dataset  # 数据集
from torch.utils.data import DataLoader  # 加载器
from torch.utils.data import random_split # 随机拆分训练集和测试集

# 1. 定义一个简单的神经网络模型
class SimpleNN(nn.Module):
    def __init__(self):
        super(SimpleNN, self).__init__()
        self.fc1 = nn.Linear(2, 2)  # 输入层到隐藏层，输入2表示有2个特征。
        self.fc2 = nn.Linear(2, 1)  # 隐藏层到输出层

    def forward(self, x):
        x = torch.relu(self.fc1(x)) # ReLU 激活函数
        x = self.fc2(x)
        return x

# 2. 创建模型实例
model = SimpleNN()

# 3. 自定义数据集
# 自定义数据集
class MyDataset(Dataset):
    def __init__(self, data, labels):
        # 数据初始化
        self.data = data
        self.labels = labels

    def __len__(self):
        # 返回数据集大小
        return len(self.data)

    def __getitem__(self, idx):
        # 按索引返回数据和标签
        sample = self.data[idx]
        label = self.labels[idx]
        return sample, label

# 4. 定义损失函数和优化器
criterion = nn.MSELoss()  # 均方误差损失函数
optimizer = optim.Adam(model.parameters(), lr=0.001)  # Adam 优化器

# 5. 假设我们有训练数据 X 和 Y
X = torch.randn(1000, 2)  # 100 个样本，2 个特征
Y = torch.randn(1000, 1)  # 100 个目标值

# 6. 实例化数据集
dataset = MyDataset(X, Y)
train_ratio = 0.8  # 训练集比例
train_size = int(train_ratio * len(dataset))
test_size = len(dataset) - train_size
train_dataset, test_dataset = random_split(dataset, [train_size, test_size])
# 实例化 DataLoader
# 其中batch_size是批量大小，也就是每次多少个数据，shuffle是是否打乱顺序，是多线程加速
train_dataloader = DataLoader(train_dataset, batch_size=10, shuffle=True, num_workers=0)
test_dataloader = DataLoader(test_dataset, batch_size=10, shuffle=True, num_workers=0)


# 7. 训练循环
# for epoch in range(100):         # 训练 100 轮
for batch_idx, (batch_data, batch_labels) in enumerate(train_dataloader):
    optimizer.zero_grad()                   # 清空之前的梯度
    output = model(batch_data)              # 前向传播
    loss = criterion(output, batch_labels)  # 计算损失
    loss.backward()                         # 反向传播
    optimizer.step()                        # 更新参数
    # 每 10 轮输出一次损失
    if (batch_idx+1) % 10 == 0:
        print(f'Epoch [{batch_idx+1}/{len(train_dataloader)}], Loss: {loss.item():.4f}')

# 8. 测试评估
# 假设你有测试集 X_test 和 Y_test
model.eval()           # 设置模型为评估模式
with torch.no_grad():  # 在评估过程中禁用梯度计算
    for batch_idx, (batch_data, batch_labels) in enumerate(test_dataloader):
        output = model(batch_data)             # 前向传播
        loss = criterion(output, batch_labels) # 计算损失
        if (batch_idx+1) % 10 == 0:
            print(f'Test Loss: {loss.item():.4f}') # 打印

```
